@app.route('/panel') def panel(): conn = get_db_connection() cursor = conn.cursor() cursor.execute(''' SELECT m.id, p.nombre, m.fecha FROM menu_del_dia m JOIN productos p ON m.producto_id = p.id ORDER BY m.fecha DESC ''') menus = cursor.fetchall() cursor.close() conn.close() return render_template('panel.html', menus=menus)
